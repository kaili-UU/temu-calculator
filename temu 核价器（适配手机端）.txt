    <h2>🔢 Temu商品核价计算器</h2>
    <div class="description">
        基于官方公式：建议销售价 = (供货价 + 物流费 + 平台扣点) ÷ (1 - 平台抽佣比例)

    <div class="input-group">

    </div>

    <div class="input-group">
        <label for="logisticsCost">物流费（元）：</label>
        <input type="number" id="logisticsCost" step="0.01" placeholder="例：0 或 6 或留空" />
    </div>

    <div class="input-group">
        <label for="platformFee">平台扣点（元）：</label>
        <input type="number" id="platformFee" step="0.01" placeholder="例：0 或 2 或留空" />
    </div>

    <div class="input-group">
        <label for="commissionRate">平台抽佣比例（如20表示20%，默认20）：</label>
        <input type="number" id="commissionRate" step="0.1" value="20" placeholder="例：20" />
    </div>

    <div class="input-group">
        <label for="minPrice">最低售价限制（元，可选）：</label>
        <input type="number" id="minPrice" step="0.01" placeholder="例：15（留空则不限制）" />
    </div>

    <button onclick="calculateSellingPrice()">💰 计算建议售价</button>

    <div class="result" id="result"></div>

    <div class="note">
        <strong>使用提示：</strong><br>
        • 支持物流费/平台扣点填写 0，也可留空（自动为 0）<br>
        • 抽佣比例如 20 表示 20%，必填项<br>
        • 若设置了最低售价且计算结果更低，将自动显示最低价<br>
        • 适配手机、平板，界面清晰易用
    </div>

    <script>
        function calculateSellingPrice() {
            const supplyPrice = parseFloat(document.getElementById('supplyPrice').value);
            const logisticsCost = parseFloat(document.getElementById('logisticsCost').value) || 0;
            const platformFee = parseFloat(document.getElementById('platformFee').value) || 0;
            const commissionRatePercent = parseFloat(document.getElementById('commissionRate').value) || 20;
            const minPriceInput = document.getElementById('minPrice').value;

            if (isNaN(supplyPrice) || supplyPrice < 0) {
                document.getElementById('result').innerText = '❌ 请输入有效的供货价';
                return;
            }

            const commissionRate = commissionRatePercent / 100;

            if (commissionRate < 0 || commissionRate >= 1) {
                document.getElementById('result').innerText = '❌ 抽佣比例应在 0% ~ 100% 之间';
                return;
            }

            const totalCost = supplyPrice + logisticsCost + platformFee;
            const denominator = 1 - commissionRate;

            if (denominator <= 0) {
                document.getElementById('result').innerText = '❌ 抽佣比例过高，无法计算';
                return;
            }

            const suggestedPrice = totalCost / denominator;

            let finalPrice = suggestedPrice;

            if (minPriceInput && !isNaN(parseFloat(minPriceInput)) && parseFloat(minPriceInput) > 0) {
                const minPrice = parseFloat(minPriceInput);
                if (suggestedPrice < minPrice) {
                    finalPrice = minPrice;
                    document.getElementById('result').innerHTML =
                        `⚠️ 计算建议价 ¥${suggestedPrice.toFixed(2)} 低于最低售价 ¥${minPrice.toFixed(2)}，<br>已自动调整为最低售价：<span style="color: #d9534f; font-size: 20px;">¥${finalPrice.toFixed(2)}</span>`;
                    return;
                }
            }

            document.getElementById('result').innerHTML =
                `✅ 建议销售价：<span style="color: #d9534f; font-size: 20px;">¥${finalPrice.toFixed(2)}</span>`;
        }
    </script>
